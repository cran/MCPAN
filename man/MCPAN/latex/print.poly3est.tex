\HeaderA{print.poly3est}{Print function for poly3est}{print.poly3est}
\keyword{print}{print.poly3est}
\begin{Description}\relax
Print function for objects of class "poly3est", as can be created by calling "poly3estf"
\end{Description}
\begin{Usage}
\begin{verbatim}

## S3 method for class 'poly3est':
print(x, digits = 4, ...)
\end{verbatim}
\end{Usage}
\begin{Arguments}
\begin{ldescription}
\item[\code{x}] An object of class "poly3est", as can be created by calling \code{\LinkA{poly3estf}{poly3estf}} 
\item[\code{digits}] a single integer, the number of digits for rounding of the results 
\item[\code{...}] currently not used... 
\end{ldescription}
\end{Arguments}
