\HeaderA{summary.binomORci}{Detailed print out for binomORci}{summary.binomORci}
\keyword{print}{summary.binomORci}
\begin{Description}\relax
Produces a more detailed print out of objects of class "binomORci", including summary statistics,
the used contrast matrix and the confidence intervals.
\end{Description}
\begin{Usage}
\begin{verbatim}
## S3 method for class 'binomORci':
summary(object, ...)
\end{verbatim}
\end{Usage}
\begin{Arguments}
\begin{ldescription}
\item[\code{object}] an object of class "binomORci" as created by function \code{\LinkA{binomORci}{binomORci}} 
\item[\code{...}] ... 
\end{ldescription}
\end{Arguments}
\begin{Value}
A print out.
\end{Value}
\begin{Author}\relax
Frank Schaarschmidt
\end{Author}
\begin{Examples}
\begin{ExampleCode}

x<-c(1,3,6,7,5)
n<-c(30,30,30,30,30)
names<-LETTERS[1:5]

ORD<-binomORci(x=x, n=n, names=names, type="Dunnett", alternative="greater")
summary(ORD)

ORW<-binomORci(x=x, n=n, names=names, type="Williams", alternative="greater")
summary(ORW)


\end{ExampleCode}
\end{Examples}
