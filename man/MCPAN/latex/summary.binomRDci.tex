\HeaderA{summary.binomRDci}{Detailed print out for binomRDci}{summary.binomRDci}
\keyword{print}{summary.binomRDci}
\begin{Description}\relax
Produces a more detailed print out of objects of class "binomRDci", including summary statistics,
the used contrast matrix and the confidence intervals.
\end{Description}
\begin{Usage}
\begin{verbatim}
## S3 method for class 'binomRDci':
summary(object, ...)
\end{verbatim}
\end{Usage}
\begin{Arguments}
\begin{ldescription}
\item[\code{object}] an object of class "binomRDci" as created by function \code{\LinkA{binomRDci}{binomRDci}} 
\item[\code{...}] further arguments to be passed to summary, currently only \code{digits} for rounding is supported 
\end{ldescription}
\end{Arguments}
\begin{Value}
A print out.
\end{Value}
