\HeaderA{summary.binomRDtest}{Detailed print out for binomRDtest}{summary.binomRDtest}
\keyword{print}{summary.binomRDtest}
\begin{Description}\relax
Produces a more detailed print out of objects of class "binomRDtest",
including summary statistics,
the used contrast matrix and the p-values.
\end{Description}
\begin{Usage}
\begin{verbatim}
## S3 method for class 'binomRDtest':
summary(object, ...)
\end{verbatim}
\end{Usage}
\begin{Arguments}
\begin{ldescription}
\item[\code{object}] an object of class "binomRDtest" as created by function \code{\LinkA{binomRDtest}{binomRDtest}}  
\item[\code{...}] further arguments to be passed to summary, currently only \code{digits} for rounding is supported 
\end{ldescription}
\end{Arguments}
\begin{Value}
A print out.
\end{Value}
